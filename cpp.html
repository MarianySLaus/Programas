            int pot1 = 0;
            int pot2 = 0;
            int resposta = 0;
            void setup()
            {
              pinMode(12, OUTPUT);
              pinMode(A1, INPUT);
              Serial.begin(9600);
              pinMode(A0, INPUT);
              pot1 = 0;
              pot2 = 0;
              resposta = 0;
              digitalWrite(12, LOW);
            }
            void loop()
            {
              pot1 += analogRead(A1);
              Serial.println("Pot1 |----------------------------------------------------------");
              Serial.println(pot1);
              pot2 += analogRead(A0);
              Serial.println("Pot2 |----------------------------------------------------------");
              Serial.println(pot2);
              if (pot1 > pot2) {
                resposta += (pot1 + pot2);
                Serial.println("Resposta |----------------------------------------------------------");
                Serial.println(resposta);
                digitalWrite(12, HIGH);
              } else {
                digitalWrite(12, LOW);
              }
              delay(1); // Wait for 1 millisecond(s)
              }
